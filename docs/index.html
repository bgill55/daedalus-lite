What's under the hood

~300 lines of TypeScript. Middleware chain. Multi-model router. No magic.

router.ts
// Route to OpenAI, Anthropic, or local models
async dispatch(request: ModelRequest): Promise {
  const model = request.model || this.config.defaultModel;

  if (model.startsWith('gpt-') || model.startsWith('o1-')) {
    return this.dispatchToOpenAI({ model, messages });
  } else if (model.startsWith('claude-')) {
    return this.dispatchToAnthropic({ model, messages });
  } else {
    return this.dispatchToLocal({ model, messages });
  }
}
middleware.ts
// Middleware chain - add logging, auth, rate limiting
use(middleware: Middleware): void {
  this.middleware.push(middleware);
}

// Error handling middleware
export const errorHandlerMiddleware = async (
  req: ModelRequest,
  next: () => Promise
): Promise => {
  try {
    return await next();
  } catch (error) {
    return {
      choices: [{
        message: { content: "Error: " + error.message }
      }]
    };
  }
};
